import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        long n = sc.nextInt();
        int num = sc.nextInt();
        int count = 0;
        
        while(n>0){
            if( n % 10 == num){
                count ++;
            }
            n = n /10;
        }
        System.out.println(count);
    }
}
