import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        int power = 1;

        while (power < num) {
            power = power * 2;
        }

        if (power == num)
            System.out.println("YES");
        else
            System.out.println("NO");

        sc.close();
    }
}
