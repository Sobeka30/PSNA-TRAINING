import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;
import java.util.Scanner;

public class Solution {

    public static void main(String args[]){
        Scanner sc = new Scanner(System.in);
        long n = sc.nextInt();
        int count = 0;
        while(n>0){
            count ++;
            n=n/10;
        }
        System.out.println(count);
    }
}
